204   def human_clock_time(seconds)
205     days = hours = mins = 0
206     if seconds >= 60 then
207       mins = ( seconds / 60 ).to_i
208       seconds = ( seconds % 60 ).to_i
209       
210       if mins >= 60 then
211         hours = ( mins / 60 ).to_i
212         mins = ( mins % 60 ).to_i
213       end
214       
215       if hours >= 24 then
216         days = ( hours / 24 ).to_i
217         hours = ( hours % 24 ).to_i
218       end
219     end
220     "#{days.to_s.rjust(3,"0")}:#{hours.to_s.rjust(2,"0")}:#{mins.to_s.rjust(2,"0")}:#{seconds.to_s.rjust(2,"0")}"
221   end

223   def fix_type(value, type)
224     # Sanitize String data into proper Class before injection 
225     if type =~ /date/ then
226       value = value.sub(/T/, ' ').sub(/Z/, '')
227       Time.parse(value)
228     elsif type =~ /integer/ then
229       value.to_i
230     elsif type =~ /string/ then
231       value.to_s
232     elsif type =~ /boolean/ then
233       if value == "true"
234         true
235       elsif value == "false"
236         false
237       else
238         raise NonBooleanValueException, "Really now? True or False?: #{value}"
239       end
240     else
241       raise UnknownDataTypeException, "I don't know how to treat type: #{type}"
242     end
243   end

365   def top_tracks_aging_well
366     results = []
367     now = Time.now
368     rating = 0
369     records = @db.execute("select rating, play_count, name, artist, play_date_utc from tracks
370                             where rating not null and play_count not null and name not null 
371                             and artist not null and play_date_utc not null
372                             order by rating, play_count")
373     records.each do |rec|
374       rating = rec[0].to_i * rec[1].to_i * Math.log( ((now - Time.parse(rec[4])) / 86400.0).to_i )
375       results.push([ rating.to_i, rec[2], rec[3], rec[4] ])
376     end
377     results.sort.reverse.first(@top)
378   end
