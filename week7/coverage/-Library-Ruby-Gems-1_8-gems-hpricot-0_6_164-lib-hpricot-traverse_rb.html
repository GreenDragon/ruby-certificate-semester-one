339     # +traverse_element+ traverses elements in the tree.
340     # It yields elements in depth first order.
341     #
342     # If _names_ are empty, it yields all elements.
343     # If non-empty _names_ are given, it should be list of universal names.
344     # 
345     # A nested element is yielded in depth first order as follows.
346     #
347     #   t = Hpricot('<a id=0><b><a id=1 /></b><c id=2 /></a>') 
348     #   t.traverse_element("a", "c") {|e| p e}
349     #   # =>
350     #   {elem <a id="0"> {elem <b> {emptyelem <a id="1">} </b>} {emptyelem <c id="2">} </a>}
351     #   {emptyelem <a id="1">}
352     #   {emptyelem <c id="2">}
353     #
354     # Universal names are specified as follows.
355     #
356     #   t = Hpricot(<<'End')
357     #   <html>
358     #   <meta name="robots" content="index,nofollow">
359     #   <meta name="author" content="Who am I?">

361     #   End
362     #   t.traverse_element("{http://www.w3.org/1999/xhtml}meta") {|e| p e}

699     # +filter+ rebuilds the tree without some components.
700     #
701     #   node.filter {|descendant_node| predicate } -> node
702     #   loc.filter {|descendant_loc| predicate } -> node
703     #
704     # +filter+ yields each node except top node.
705     # If given block returns false, corresponding node is dropped.
706     # If given block returns true, corresponding node is retained and
707     # inner nodes are examined.
708     #
709     # +filter+ returns an node.
710     # It doesn't return location object even if self is location object.
711     #
712     def filter(&block)
713       subst = {}
714       each_child_with_index {|descendant, i|
715         if yield descendant
716           if descendant.elem?
717             subst[i] = descendant.filter(&block)
718           else
719             subst[i] = descendant
720           end
721         else
722           subst[i] = nil
723         end
724       }
725       to_node.subst_subnode(subst)
726     end

745     # +author+ searches author and return it as a text.
746     # It returns nil if not found.
747     #
748     # +author+ searchs following information.
749     #
750     # - <meta name="author" content="author-name"> in HTML
751     # - <link rev="made" title="author-name"> in HTML
752     # - <dc:creator>author-name</dc:creator> in RSS
753     # - <dc:publisher>author-name</dc:publisher> in RSS
754     def author
755       traverse_element('meta',
756         '{http://www.w3.org/1999/xhtml}meta') {|e|
757         begin
758           next unless e.fetch_attr('name').downcase == 'author'
759           author = e.fetch_attribute('content').strip
760           return author if !author.empty?
761         rescue IndexError
762         end
763       }
764 
765       traverse_element('link',
766         '{http://www.w3.org/1999/xhtml}link') {|e|
767         begin
768           next unless e.fetch_attr('rev').downcase == 'made'
769           author = e.fetch_attribute('title').strip
770           return author if !author.empty?
771         rescue IndexError
772         end
773       } 
774 
775       if channel = find_element('{http://purl.org/rss/1.0/}channel')
776         channel.traverse_element('{http://purl.org/dc/elements/1.1/}creator') {|e|
777           begin
778             author = e.extract_text.strip
779             return author if !author.empty?
780           rescue IndexError
781           end
782         }
783         channel.traverse_element('{http://purl.org/dc/elements/1.1/}publisher') {|e|
784           begin
785             author = e.extract_text.strip
786             return author if !author.empty?
787           rescue IndexError
788           end
789         }
790       end
